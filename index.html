// answer[i] == "FizzBuzz" if i is divisible by 3 and 5.

let result = [];
for (let i = 1; i <= n; i++) {
  var inarray = "";
  if (i % 3 == 0 && i % 5 == 0) {
    inarray += "FizzBuzz"; //array will get FizzBuzz
  }
  if (i % 3 == 0) {
    inarray += "Fizz"; //array will get Fizz
  }
  if (i % 5 == 0) {
    inarray += "Buzz"; //array will get Fizz
  }
  if (inarray === "") {
    inarray += i; //array will get number
  }
  result.push(inarray);
}

function bubbleSort(Array) {
  let len = Array.length;
  for (let i = 0; i < len; i++) {
    //you can also use "for in", so you don't need the variable "len"
    for (let j = 0; j < len; j++) {
      if (Array[j] > Array[j + 1]) {
        let tmp = Array[j];
        Array[j] = Array[j + 1];
        Array[j + 1] = tmp;
      }
    }
  }
  return Array;
}

/* 
week6-Day5-Assignment

Input: digits = [1,2,3]
Output: [1,2,4]
Explanation: The array represents the integer 123.
Incrementing by one gives 123 + 1 = 124.
Thus, the result should be [1,2,4]. 

*/

function simple(digits) {
  if (!digits || digits.length == 0) return [];
  for (let i = digits.length - 1; i >= 0; i--) {
    if (digits[i] < 9) {
      //998->999  989->990
      digits[i]++; //units place digit is
      return digits;
    } else {
      digits[i] = 0;
    }
  }
  //only special case 999-> 1000
  let thousand = new Array(digits.length).fill(0);
  thousand.unshift(1); //add 1 in head of arr
  return thousand;
}

digits = [6, 1, 4, 5, 3, 9, 0, 1, 9, 5, 1, 8, 6, 7, 0, 5, 5, 4, 3, 7, 8];

console.log(simple(digits));

var reverseString = function (s) {
  if (s.length == 0) {
    return;
  }
  var temp = s[0];
  s.shift();
  reverseString(s);
  s.push(temp);
  return s; //return the reverse string
};

/* 
week10 day5 assignment Q.2
The Fibonacci numbers, commonly denoted F(n) form a sequence, called the Fibonacci sequence, 
such that each number is the sum of the two preceding ones, starting from 0 and 1. That is,

F(0) = 0, F(1) = 1
F(n) = F(n - 1) + F(n - 2), for n > 1.
Given n, calculate F(n).
*/
var fib = function (n) {
  if (n == 0) return 0;
  if (n == 1) return 1;
  return fib(n - 1) + fib(n - 2);
};

console.log(fib(7)); //finding the 7th term of fib sequence which is 13

/* 
----Output----
13
*/

/* 
LC-88. Merge Sorted Array

Input: nums1 = [1,2,3,0,0,0], m = 3, nums2 = [2,5,6], n = 3
Output: [1,2,2,3,5,6]
Explanation: The arrays we are merging are [1,2,3] and [2,5,6].
The result of the merge is [1,2,2,3,5,6] with the underlined elements coming from nums1.
 */
var merge = function (nums1, m, nums2, n) {
  let index1 = m - 1;
  let index2 = n - 1;

  for (let i = n + m - 1; i >= 0 && index2 >= 0; i--) {
    const num1 = nums1[index1];
    const num2 = nums2[index2];

    if (index1 >= 0 && num1 > num2) {
      nums1[i] = num1;
      index1--;
    } else {
      nums1[i] = num2;
      index2--;
    }
  }
  return nums1;
};

console.log("Result after Merge: " + merge([2, 4, 6, 8, 10, 13], 2, [7, 9, 15, 49, 63, 72], 4));
/* 
Result after Merge: 2,4,7,9,15,49
 */
